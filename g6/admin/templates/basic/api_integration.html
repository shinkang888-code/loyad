연동 예시 코드

        
아래 예시에서 {{ api_base }}/ 부분을 로컬/배포 URL로 바꿔 사용하세요.

        

            
            
            
        

        

            

                토큰 발급
                
            

            
curl -X POST "{{ api_base }}/token" \
  -H "Content-Type: application/x-www-form-urlencoded" \
  -d "username=아이디&password=비밀번호"

            

                인증 API 호출 (메뉴 목록)
                
            

            
curl -H "Authorization: Bearer [발급받은_Access_Token]" \
  "{{ api_base }}/menus"

        

        

            
// 1. 토큰 발급
const tokenRes = await fetch('{{ api_base }}/token', {
  method: 'POST',
  headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
  body: 'username=아이디&password=비밀번호'
});
const { access_token } = await tokenRes.json();

// 2. API 호출
const res = await fetch('{{ api_base }}/menus', {
  headers: { 'Authorization': `Bearer ${access_token}` }
});
const data = await res.json();

            
        

        

            
import requests

BASE = "{{ api_base }}"

# 1. 토큰 발급
r = requests.post(f"{BASE}/token", data={"username": "아이디", "password": "비밀번호"})
token = r.json()["access_token"]

# 2. API 호출
resp = requests.get(f"{BASE}/menus", headers={"Authorization": f"Bearer {token}"})
data = resp.json()
